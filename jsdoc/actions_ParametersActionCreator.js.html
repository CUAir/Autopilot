// @flow

import GCSDispatcher from 'js/dispatcher/GCSDispatcher';
import { ParametersTypes } from 'js/constants/ActionTypes';

/**
 * @module actions/ParametersActionCreator
 */

/**
 * Sets a single parameter
 * @param {string} name
 * @param {NValue} value
 * @returns {undefined}
 */
export function receiveSingleParameter(name: string, value: NValue): void {
  GCSDispatcher.dispatch({
    type: ParametersTypes.RECEIVE_SINGLE_PARAMETER,
    name, value
  });
}

/**
 * Sets multiple parameters
 * @param {Parameter[]} parameters
 * @returns {undefined}
 */
export function receiveMultipleParameters(parameters: Parameter[]): void {
  GCSDispatcher.dispatch({
    type: ParametersTypes.RECEIVE_MULTIPLE_PARAMETERS,
    parameters
  });
}

/**
 * Sets the max parameter count
 * @param {number} count
 * @returns {undefined}
 */
export function receiveMaxParamCount(count: number): void {
  GCSDispatcher.dispatch({
    type: ParametersTypes.RECEIVE_MAX_PARAM_COUNT,
    count
  });
}

/**
 * Edits the value of a parameter
 * @param {string} name
 * @param {NValue} value
 * @returns {undefined}
 */
export function editParameter(name: string, value: NValue): void {
  GCSDispatcher.dispatch({
    type: ParametersTypes.EDIT_PARAMETER,
    name, value
  });
}

/**
 * Edits list of parameters
 * @param {Parameter[]} parameters
 * @returns {undefined}
 */
export function loadParametersFromFile(parameters: Parameter[]): void {
  GCSDispatcher.dispatch({
    type: ParametersTypes.PARAMETER_LOAD_FROM_FILE,
    parameters
  });
}

/**
 * Sets a parameter as selected
 * @param {string} name
 * @returns {undefined}
 */
export function selectParameter(name: string): void {
  GCSDispatcher.dispatch({
    type: ParametersTypes.SELECT_PARAMETER,
    name
  });
}

/**
 * Changes the note value in the store/display
 * Used when downloading parameters
 * @param {boolean} note
 * @returns {undefined}
 */
export function receiveNote(note: string): void { 
  GCSDispatcher.dispatch({
    type: ParametersTypes.RECEIVE_NOTE,
    note
  });
}

/**
 * Changes the value of the parameter search field
 * @param {string} search_field
 * @returns {undefined}
 */
export function receiveSearchField(search_field: string): void {
  GCSDispatcher.dispatch({
    type: ParametersTypes.RECEIVE_SEARCH_FIELD,
    search_field
  });
}

/**
 * Changes whether the parameter modal is open
 * @param {boolean} open
 * @returns {undefined}
 */
export function receiveModalOpen(open: boolean): void {
  GCSDispatcher.dispatch({
    type: ParametersTypes.RECEIVE_MODAL_OPEN,
    open
  });
}

/**
 * Tells the parameter store to save all of the selected parameters
 * @returns {undefined}
 */
export function saveParamModal(): void {
  GCSDispatcher.dispatch({
    type: ParametersTypes.PARAMETER_SAVE_FROM_MODAL
  });
}

// @flow

import GCSDispatcher from 'js/dispatcher/GCSDispatcher';
import { WaypointTypes } from 'js/constants/ActionTypes';

/**
 * @module actions/WaypointActionCreator
 */

/**
 * Updates a waypoint
 * @param {Waypoint} start
 * @param {Waypoint} end
 * @returns {undefined}
 */
export function updateItem(start: Waypoint, end: Waypoint): void {
  GCSDispatcher.dispatch({
    type: WaypointTypes.REORDER_WAYPOINTS,
    start, end
  });
}

/**
 * Updates a cell of a waypoint to a new value
 * @param {number} number
 * @param {string} key
 * @param {NValue} newValue
 * @returns {undefined}
 */
export function cellChange(number: number, key: string, newValue: NValue): void {
  GCSDispatcher.dispatch({
    type: WaypointTypes.WAYPOINT_UPDATE_CELL,
    number, key, newValue
  });
}

/**
 * Updates the lat and lon of a waypoint
 * @param {number} index
 * @param {number} lat
 * @param {number} lon
 * @returns {undefined}
 */
export function updateCellLatLon(index: number, lat: number, lon: number): void {
  GCSDispatcher.dispatch({
    type: WaypointTypes.WAYPOINT_UPDATE_CELL_LAT_LON,
    index, lat, lon
  });
}

/**
 * Confirms that the user wanted to click on the waypoint in the list
 * Warns if waypoint is already sent (changes are immediately pushed)
 * @param {number} index
 * @param {Object} formRef - References to form in DOM
 * @returns {undefined}
 */
export function confirmSelect(index: number, formRef: Object): void {
  GCSDispatcher.dispatch({
    type: WaypointTypes.WAYPOINT_CONFIRM_SELECT,
    index, formRef
  });
}

/**
 * Confirms that the user wanted to click on the waypoint in the map
 * Warns if waypoint is already sent (changes are immediately pushed)
 * @param {number} index - index of waypoint in list
 * @returns {undefined}
 */
export function confirmSelectMap(index: number): void {
  GCSDispatcher.dispatch({
    type: WaypointTypes.WAYPOINT_CONFIRM_SELECT_MAP,
    index
  });
}

/**
 * Sets the waypoint as selected
 * @param {number} number
 * @returns {undefined}
 */
export function setSelected(number: number): void {
  GCSDispatcher.dispatch({
    type: WaypointTypes.WAYPOINT_SET_SELECTED,
    number
  });
}

/**
 * Adds a temporary waypoint
 * @param {number} lat
 * @param {number} lon
 * @returns {undefined}
 */
export function addWaypoint(lat: number, lon: number): void {
  GCSDispatcher.dispatch({
    type: WaypointTypes.WAYPOINT_ADD,
    lat, lon
  });
}

/**
 * Sends the selected waypoint to the server
 * @returns {undefined}
 */
export function sendWaypoint(): void {
  GCSDispatcher.dispatch({
    type: WaypointTypes.WAYPOINT_SEND
  });
}

/**
 * Deletes the selected waypoint
 * @returns {undefined}
 */
export function deleteWaypoint(): void {
  GCSDispatcher.dispatch({
    type: WaypointTypes.WAYPOINT_DELETE
  });
}

/**
 * Sets the selected waypoint as current
 * @returns {undefined}
 */
export function setCurrent(): void {
  GCSDispatcher.dispatch({
    type: WaypointTypes.WAYPOINT_SET_CURRENT
  });
}

 /**
 * Loads waypoints from file
 * @param {Waypoint[]} wps
 * @returns {undefined}
 */
export function loadWaypointsFromFile(wps: Waypoint[]): void {
  GCSDispatcher.dispatch({
    type: WaypointTypes.WAYPOINTS_LOAD_FROM_FILE,
    wps
  });
}

/**
 * Sets the waypoints to the new batch of received waypoints
 * @param {Waypoint[]} waypoints
 * @returns {undefined}
 */
export function receiveWaypoints(waypoints: Waypoint[]): void {
  GCSDispatcher.dispatch({
    type: WaypointTypes.WAYPOINTS_RECEIVE,
    waypoints
  });
}

/**
 * Sets the received waypoint number as current
 * @param {number} current
 * @returns {undefined}
 */
export function receiveCurrent(current: number): void {
  GCSDispatcher.dispatch({
    type: WaypointTypes.WAYPOINTS_RECEIVE_CURRENT,
    current
  });
}

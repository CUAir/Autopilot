// @flow

import React from 'react';

import StatusStore from 'js/stores/StatusStore';
import WaypointStore from 'js/stores/WaypointStore';
import InteropStore from 'js/stores/InteropStore';
import ParametersStore from 'js/stores/ParametersStore';

import FullWidthButton from 'js/components/General/FullWidthButton';

import { valueToColor, colorToButton, valueToColorName } from 'js/utils/ComponentUtils';

/**
 * @module components/Navbar/GoNoGoDropdown
 */

function getStateFromStore() {
  let goNoGoList = [];
  goNoGoList = goNoGoList.concat(InteropStore.getGoNoGo());
  goNoGoList = goNoGoList.concat(ParametersStore.getGoNoGo());
  goNoGoList = goNoGoList.concat(WaypointStore.getGoNoGo());
  goNoGoList = goNoGoList.concat(StatusStore.getGoNoGo());
  return {goNoGo: goNoGoList};
}

function buildFalseArray(length) {
  let arr = [];
  for (let i = 0; i < length; i++) {
    arr.push(false);
  }
  return arr;
}

const GoNoGoDropdown = React.createClass({
  getInitialState: function() {
    let storedArray = localStorage['goNoGo'];
    let checkArr = [];
    const goNoGoLength = getStateFromStore().goNoGo.length;
    if (storedArray == undefined || JSON.parse(storedArray).length < goNoGoLength) {
      checkArr = buildFalseArray(goNoGoLength);
    } else {
      checkArr = JSON.parse(storedArray);
    }
    localStorage['goNoGo'] = JSON.stringify(checkArr);
    return Object.assign(
      {},
      getStateFromStore(),
      {checked: checkArr}
    );
  },

  componentDidMount: function() {
    StatusStore.addChangeListener(this._onChangeStatus);
    WaypointStore.addChangeListener(this._onChangeStatus);
    InteropStore.addChangeListener(this._onChangeStatus);
    ParametersStore.addChangeListener(this._onChangeStatus);
  },

  componentWillUnmount: function() {
    StatusStore.removeChangeListener(this._onChangeStatus);
    WaypointStore.removeChangeListener(this._onChangeStatus);
    InteropStore.removeChangeListener(this._onChangeStatus);
    ParametersStore.removeChangeListener(this._onChangeStatus);
  },

  _onChangeStatus: function() {
    this.setState(getStateFromStore());
  },

  _onChangeClick: function(i: number) {
    let currentstate = this.state.checked;
    currentstate[i] = !currentstate[i];
    this.setState({checked: currentstate});
    localStorage['goNoGo'] = JSON.stringify(currentstate);
  },

  createCheckBox: function(item: Object, i: number) {
    const val = item.autoCheck ? item.value : this.state.checked[i];
    return (
      <tr key={item.key}>
        <td>
          <input disabled={item.autoCheck} type={item.type} checked={val} onChange={this._onChangeClick.bind(this, i)} />
        </td>
        <td>{item.name}</td>
      </tr>
    );
  },

  goNoGo: function(dropdown) {
    const v = this.goCheck() ? 100: 0;
    const className = `${valueToColor(v)} ${colorToButton(valueToColor(v))}`;
    const color = valueToColorName(v);
    const outerstyle = {height: '50px', marginBottom: '0px'};
    return (
      <li key="GO_NOGO" className={className} id="GO_NOGO" style={outerstyle}>
        <a data-toggle="dropdown" style={{backgroundColor: color, color: 'white'}}>
          GO_NOGO<span className="caret"></span>
        </a>
        {dropdown}
      </li>
    );
  },

  /**
  * checks if all the boxes are checked
  * reaturns a boolean representing whether the boxes are all checked
  */
  goCheck() {
    const list = this.state.goNoGo;
    const checklist = this.state.checked;
    for (let i = 0; i < list.length; i++) {
      if (list[i].autoCheck) {
        if (!list[i].value) {
          return false;
        }
      } else {
        if (!checklist[i]) {
          return false;
        }
      }
    }
    return true;
  },

  _clearStorage: function() {
    const newArr = buildFalseArray(getStateFromStore().goNoGo.length);
    this.setState({checked: newArr});
    localStorage['goNoGo'] = JSON.stringify(newArr);
  },

  render: function() {
    return this.goNoGo(
      <ul className='dropdown-menu' style={{width: '350px', color: 'black'}}>
        <table className="table"><tbody>
          {this.state.goNoGo.map(this.createCheckBox)}
          <FullWidthButton color="error" onClick={this._clearStorage} text="Clear Checklist" />
        </tbody></table>
      </ul>
    );
  }
});

export default GoNoGoDropdown;

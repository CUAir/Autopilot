import GCSDispatcher from 'js/dispatcher/GCSDispatcher';
import { StatusTypes } from 'js/constants/ActionTypes';

 * @module actions/StatusActionCreator
 */

/**
 * Changes airspeed values
 * @param {Airspeed} airspeed
 * @returns {undefined}
 */
export function receiveAirspeed(airspeed: Airspeed): void {
  GCSDispatcher.dispatch({
    type: StatusTypes.RECEIVE_AIRSPEED,
    airspeed
  });
}

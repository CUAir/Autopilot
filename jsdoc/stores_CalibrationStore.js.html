// @flow

import GCSDispatcher from 'js/dispatcher/GCSDispatcher';
import { CalibrationTypes } from 'js/constants/ActionTypes';
import { BUTTON } from 'js/constants/ButtonTypes';
import Store from 'js/stores/Store';

import { clearCalibrationLines } from 'js/actions/CalibrationActionCreator';

import { calibrationAStart, calibrationGStart, calibrationPStart, calibrationContinue,
         calibrationRCStart, calibrationRCStop } from '../utils/SendApi';

/**
 * @module stores/CalibrationStore
 */

let _lines: string[] = [];

const _calibration_controls: FullWidthElementType[] = [
  {color: 'success', text: 'Start Accelerometer Calibration', type: BUTTON, onClick: calibrationAStart, key: 'CALIBRATION_A_START'},
  {color: 'success', text: 'Start Gyroscope Calibration', type: BUTTON, onClick: calibrationGStart, key: 'CALIBRATION_G_START'},
  {color: 'success', text: 'Start Pressure Calibration', type: BUTTON, onClick: calibrationPStart, key: 'CALIBRATION_P_START'},
  {color: 'success', text: 'Start RC Calibration', type: BUTTON, onClick: calibrationRCStart, key: 'CALIBRATION_RC_START'},
  {color: 'error', text: 'Stop RC Calibration', type: BUTTON, onClick: calibrationRCStop, key: 'CALIBRATION_RC_STOP'},
  {color: 'unknown', text: 'Continue', type: BUTTON, onClick: calibrationContinue, key: 'CALIBRATION_CONTINUE'},
  {color: 'warning', text: 'Clear', type: BUTTON, onClick: clearCalibrationLines, key: 'CALIBRATION_CLEAR'}
];

function _add_line(line: string): void { _lines.push(line.replace('any key', 'continue')); }
function _clear_lines(): void { _lines = []; }

class CalibrationStore extends Store {

  constructor() {
    super();
  }

  /**
   * Returns the calibration output lines
   * @returns {string[]}
   */
  getLines(): string[] {
    return _lines;
  }

  /**
   * Returns the calibration controls
   * @returns {FullWidthElementType[]}
   */
  getControls(): FullWidthElementType[] {
    return _calibration_controls;
  }
}

const calibrationStoreInstance = new CalibrationStore();

calibrationStoreInstance.dispatchToken = GCSDispatcher.register(action => {
  switch (action.type) {
  case CalibrationTypes.ADD_CALIBRATION_LINE:
    _add_line(action.line);
    calibrationStoreInstance.emitChange();
    break;
  case CalibrationTypes.CLEAR_CALIBRATION:
    _clear_lines();
    calibrationStoreInstance.emitChange();
    break;
  default:
    // do nothing
  }
});

export default calibrationStoreInstance;

// @flow

import React from 'react';
import $ from 'jquery';

import ModeButtonElement from 'js/components/General/ModeButtonElement';

/**
 * @module components/PlaneAction/ModeButton
 */

const ModeButton = React.createClass({
  propTypes: {
    name: React.PropTypes.string.isRequired,
    key: React.PropTypes.string,
    id: React.PropTypes.string,
    value: React.PropTypes.string.isRequired,
    modes: React.PropTypes.arrayOf(React.PropTypes.object).isRequired
  },

  /** Returns a mode button element
   */
  getModeButtonElement: function(item: DropdownOptionType) {
    const onClick = typeof item.onClick === 'function' ? item.onClick : $.noop;
    const boundClick = onClick.bind(this, this.props.name, item.name);
    return <ModeButtonElement name={item.name} onClick={boundClick} key={item.key} />;
  },

  render: function() {
    return (
      <tr id={this.props.id}>
        <td>
          <p className='middle-center'>{this.props.name}</p>
        </td>
        <td>
          <div className='dropup full'>
            <button data-toggle='dropdown' key={this.props.key} className='btn btn-primary dropdown-toggle full middle-center'>
              {this.props.value}<span className='caret'></span>
            </button>
            <ul className='dropdown-menu'>{this.props.modes.map(this.getModeButtonElement)}</ul>
          </div>
        </td>
      </tr>
    );
  }
});

export default ModeButton;

    // @flow

import React from 'react';
import autobind from 'autobind-decorator';

import ParametersItem from 'js/components/Parameters/ParametersItem';

import ParametersStore from 'js/stores/ParametersStore';

import * as ParametersActionCreator from 'js/actions/ParametersActionCreator';

function getStateFromStore() {
  return {
    parameters: ParametersStore.getSearchedParameters()
  };
}

function getParametersItem(item: Object) {
  function _onValChange(e) {
    const val = !isNaN(parseFloat(e)) ? e : e.target.value;

    if (item.range !== undefined) {
      const range = item.range.split(' ').map(parseFloat);
      if ((val > range[1] && range[1] > 0) || (val < range[0] && range[0] < 0)) {
        return;
      }
    }

    if (item.increment !== undefined) {
      const divided = val / parseFloat(item.increment);
      if (parseInt(divided) !== divided && 
        !isNaN(parseFloat(item.increment))) {
        return;
      }
    }

    ParametersActionCreator.editParameter(item.name, val);
    return;
  }

  let color;
  if (ParametersStore.getSelected() === item.name) {
    color = '#5CB85C';
  } else if (item.value === '') {
    color = '#A9A9A9';
  } else {
    color = 'white';
  }

  return (
    <ParametersItem
      key={item.key}
      name={item.name}
      description={item.description}
      value={item.value}
      units={item.units}
      values={item.values}  
      documentation={item.documentation}
      isTemp={item.isTemp}
      color={color}
      onChange={_onValChange}
      units={item.units}
      id={item.name}
      onClick={ParametersActionCreator.selectParameter.bind(this, item.name)} />
  );
}

export default class ParametersTableSection extends React.Component {

  state = getStateFromStore();

  constructor(props: Object) {
    super(props);
  }

  componentDidMount() {
    ParametersStore.addChangeListener(this._onChange);
  }

  componentWillUnmount() {
    ParametersStore.removeChangeListener(this._onChange);
  }

  @autobind
  _onChange() {
    this.setState(getStateFromStore());
  }

  render() {
    return (
      <div id='parametersDiv' className='col-xs-12 col-md-10'>
        <table className='table' id='parametersTable'>
          <thead>
            <tr>
              <th>
                <strong>Parameters</strong>
              </th>
            </tr>
            <tr>
              <th>Name</th>
              <th>Description</th>
              <th>Value</th>
              <th>Documentation</th>
            </tr>
          </thead>
          <tbody className='full'>
            {this.state.parameters.map(getParametersItem)}
          </tbody>
        </table>
      </div>
    );
  }
}

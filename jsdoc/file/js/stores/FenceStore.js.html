    js/stores/FenceStore.js

    // @flow

import GCSDispatcher from 'js/dispatcher/GCSDispatcher';
import { FenceTypes } from 'js/constants/ActionTypes';
import { CHECKBOX, FILE_INPUT, BUTTON } from 'js/constants/ButtonTypes';
import Store from 'js/stores/Store';

import { receiveFencesEnabled } from 'js/actions/FenceActionCreator';

import { loadFenceFromFile } from 'js/utils/LoadApi';
import { saveFenceToFile } from 'js/utils/DownloadApi';

let _fence_points: Point[] = [];
let _fences_enabled: boolean = true;

function _update_fence_points(fence_points: Point[]): void { _fence_points = fence_points; }
function _update_fences_enabled(fences_enabled: boolean): void { _fences_enabled = fences_enabled; }

function fencesChange(e: EventObject): void {
  receiveFencesEnabled(e.nativeEvent.target.checked);
}

function saveFence(): void {
  saveFenceToFile(_fence_points);
}

class FenceStore extends Store {

  constructor() {
    super();
  }

  /**
   * Returns whether fences is checked
   * @returns {boolean}
   */
  getFencesEnabled(): boolean {
    return _fences_enabled;
  }

  /**
   * Returns the fence points
   * @returns {Point[]}
   */
  getFencePoints(): Point[] {
    return _fence_points;
  }

  /**
   * Returns the settings buttons for fences
   * @returns {SettingsItemType[]}
   */
  getButtons(): SettingsItemType[] {
    return [
      {name: 'Fences', value: _fences_enabled, id: 'FENCES', key: 'FENCES', type: CHECKBOX, onChange: fencesChange},
      {name: 'Load Geofence', id: 'LOAD_FENCES', key: 'LOAD_FENCES', type: FILE_INPUT, onChange: loadFenceFromFile},
      {name: 'Save Geofence', id: 'SAVE_FENCES', key: 'SAVE_FENCES', type: BUTTON, onClick: saveFence}
    ];
  }
}

const fenceStoreInstance = new FenceStore();

fenceStoreInstance.dispatchToken = GCSDispatcher.register(action => {
  switch(action.type) {
  case FenceTypes.RECEIVE_FENCE_POINTS:
    _update_fence_points(action.points);
    fenceStoreInstance.emitChange();
    break;
  case FenceTypes.RECEIVE_FENCES_ENABLED:
    _update_fences_enabled(action.fences);
    fenceStoreInstance.emitChange();
    break;
  default:
    // do nothing
  }
});

export default fenceStoreInstance;

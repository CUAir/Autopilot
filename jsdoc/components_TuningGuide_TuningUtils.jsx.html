// @flow

import React from 'react'; // eslint-disable-line no-unused-vars

import TuningRow from 'js/components/TuningGuide/TuningRow';

import { sendParameter } from 'js/utils/SendApi';

import * as TuningActionCreator from 'js/actions/TuningActionCreator';

/**
 * @module components/TuningGuide/TuningUtils
 */

/**
 * Sends the parameter to the backend and updates the value in the TuningGuide
 * @param {string} name
 * @param {NValue} value
 * @param {number} index
 * @param {number} flightNumber
 * @returns {function}
 */
export function sendParam(name: string, value: NValue, index: number, flightNumber: number) {
  return () => {
    sendParameter({name, value});
    TuningActionCreator.updateValue(flightNumber, index, true);
  };
}

/**
 * Builds a tuning row
 * @param {Object} item
 * @param {number} index
 * @return {Component}
 */
export function buildTuningRow(item: Object, index: number) {
  return <TuningRow index={index} {...item} />;
}

/**
 * Sends the text as a param to the backend and updates the value in the TuningGuide
 * @param {string} name
 * @param {string} text
 * @param {number} index
 * @param {number} flightNumber
 * @returns {function}
 */
export function sendTextParam(name: string, text: string, index: number, flightNumber: number) {
  return () => {
    sendParameter({name, value: text});
    TuningActionCreator.updateValue(flightNumber, index, {text: text, value: true});
  };
}

/**
 * Changes the text value in the TuningGuide
 * @param {string} name
 * @param {number} index
 * @param {number} flightNumber
 */
export function changeTextParam(name: string, index: number, flightNumber: number) {
  return (text: NValue) => {
    TuningActionCreator.updateValue(flightNumber, index, {text, value: false});
  };
}
